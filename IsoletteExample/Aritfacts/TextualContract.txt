functions
    //REGULATOR

        //END-TO-END Regulator Properties

            //----------------------------------------------
            //  Property:  CT < LDT implies Heat-Control ON
            //    [high-level]
            //      In Normal mode, and in the absence of error-triggering inputs,
            //      If current temp is less than lower desired, then heat control shall be ON
            //
            //    Note: Properties should be traceable to system requirements
            //----------------------------------------------

            def sysProp_NormalModeHeatOnn( regulator_mode: Isolette_Data_Model::Regulator_Mode,
                                        current_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                        lower_desired_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                        upper_desired_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                        internal_failure: Isolette_Data_Model::Failure_Flag_i,
                                        heat_control: Isolette_Data_Model::On_Off
                                        ): Base_Types::Boolean :=
                (lower_desired_tempWstatus.status == Isolette_Data_Model::ValueStatus.Valid
                and upper_desired_tempWstatus.status == Isolette_Data_Model::ValueStatus.Valid
                and current_tempWstatus.status == Isolette_Data_Model::ValueStatus.Valid
                and regulator_mode == Isolette_Data_Model::Regulator_Mode.Normal_Regulator_Mode
                and current_tempWstatus.degrees < lower_desired_tempWstatus.degrees)
                implies (heat_control == Isolette_Data_Model::On_Off::Onn)

        //MRI property functions

            //----------------------------------------------
            //  derived from REQ_MRI_7
            //
            //  If the Status attribute of the Lower Desired Temperature
            //  and the Upper Desired Temperature is Valid,
            //  the Regulator Interface Failure shall be set to False.
            //  https://www.faa.gov/sites/faa.gov/files/aircraft/air_cert/design_approvals/air_software/AR-08-32.pdf#page=108 "
            //----------------------------------------------
        
            def sysProp_REQ_MRI_7( lower_desired_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                upper_desired_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                interface_failure: Isolette_Data_Model::Failure_Flag_i
                                ): Base_Types::Boolean :=
                (interface_failure.flag == (not((upper_desired_tempWstatus.status == Isolette_Data_Model::ValueStatus.Valid) 
                                            and (lower_desired_tempWstatus.status == Isolette_Data_Model::ValueStatus.Valid))))
            
            //----------------------------------------------
            //derived from REQ_MRI_8
            //
            //If the Regulator Interface Failure is False,
            //the Desired Range shall be set to the Desired Temperature Range.
            //https://www.faa.gov/sites/faa.gov/files/aircraft/air_cert/design_approvals/air_software/AR-08-32.pdf#page=108 "
            //----------------------------------------------

            def sysProp_REQ_MRI_8( lower_desired_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                upper_desired_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                lower_desired_temp: Isolette_Data_Model::Temp_i,
                                upper_desired_temp: Isolette_Data_Model::Temp_i,
                                interface_failure: Isolette_Data_Model::Failure_Flag_i
                                ): Base_Types::Boolean :=
                        (not interface_failure.flag)
                        imples (lower_desired_temp.degrees == lower_desired_tempWstatus.degrees 
                                and upper_desired_temp.degrees == upper_desired_tempWstatus.degrees)

        //MRI Assumption
       
            //derived from lower_is_not_higher_than_upper from MRI
            def sysProp_lower_is_not_higher_than_upper (lower_desired_tempWstatus: Isolette_Data_Model::TempWstatus_i, 
                                                        upper_desired_tempWstatus: Isolette_Data_Model::TempWstatus_i): Base_Types::Boolean :=
                    lower_desired_tempWstatus.degrees <= upper_desired_tempWstatus.degrees

        //MHS Assumption

            //derived from lower_is_lower_temp from MHS
            def sysProp_lower_is_lower_temp (lower_desired_temp: Isolette_Data_Model::Temp_i, 
                                            upper_desired_temp: Isolette_Data_Model::Temp_i): Base_Types::Boolean :=
                    lower_desired_temp.degrees <= upper_desired_temp.degrees

    //MONITOR
        //Monitor End-to-End property

            //----------------------------------------------
            //  Property:  CT < LAT || CT > UAT implies alarm_control ON
            //    [high-level]
            //      In Normal mode, and in the absence of error-triggering inputs,
            //      If current temp is less than lower alarm temp or greater that
            //      the upper alarm temp, then heat control shall be ON
            //----------------------------------------------

            def sysProp_NormalModeAlarmOn( 
                                            lower_alarm_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                            upper_alarm_tempWStatus: Isolette_Data_Model::TempWstatus_i,
                                            current_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                            monitor_mode: Isolette_Data_Model::Monitor_Mode
                                            interface_failure: Isolette_Data_Model::Failure_Flag_i,
                                            alarm_control: Isolette_Data_Model::On_Off
                                        ): Base_Types::Boolean :=
                        (lower_alarm_tempWstatus.status == Isolette_Data_Model::ValueStatus.Valid
                            and upper_alarm_tempWStatus.status == Isolette_Data_Model::ValueStatus.Valid
                            and current_tempWstatus.status == Isolette_Data_Model::ValueStatus.Valid    
                            and monitor_mode == Isolette_Data_Model::Monitor_Mode::Normal_Monitor_Mode &&
                            and (current_tempWstatus.degrees < lower_alarm_tempWstatus.degrees
                                or current_tempWstatus.degrees > upper_alarm_tempWStatus.degrees))
                            implies alarm_control == Isolette_Data_Model::On_Off::Onn

        //MMI property functions

            //----------------------------------------------
            // If the Status attribute of the Lower Alarm Temperature
            // and the Upper Alarm Temperature is Valid,
            // the Monitor Interface Failure shall be set to False
            // https://www.faa.gov/sites/faa.gov/files/aircraft/air_cert/design_approvals/air_software/AR-08-32.pdf#page=113 
            //
            // derived from REQ_MMI_5
            //----------------------------------------------


            def sysProp_REQ_MMI_5(lower_alarm_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                upper_alarm_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                interface_failure: Isolette_Data_Model::Failure_Flag_i
                                ): Base_Types::Boolean :=
                        ((upper_alarm_tempWstatus.status == Isolette_Data_Model::ValueStatus.Valid) 
                            and (lower_alarm_tempWstatus.status == Isolette_Data_Model::ValueStatus.Valid))
                        implies (not interface_failure.flag)

            //----------------------------------------------
            // If the Monitor Interface Failure is False,
            // the Alarm Range variable shall be set to the Desired Temperature Range
            // https://www.faa.gov/sites/faa.gov/files/aircraft/air_cert/design_approvals/air_software/AR-08-32.pdf#page=113
            //
            // derived from REQ_MMI_6
            //----------------------------------------------

            def sysProp_REQ_MMI_6(lower_alarm_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                            upper_alarm_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                            lower_alarm_temp: Isolette_Data_Model::Temp_i,
                            upper_alarm_temp: Isolette_Data_Model::Temp_i,
                            interface_failure: Isolette_Data_Model::Failure_Flag_i
                            ): Base_Types::Boolean :=
                    (not interface_failure.flag)
                    imples (lower_alarm_temp.degrees == lower_alarm_tempWstatus.degrees 
                        and upper_alarm_temp.degrees == upper_alarm_tempWstatus.degrees)

        //MMI Integration Constraints

            //derived from Allowed_LowerAlarmTemp from MMI
            def sysProp_I_Assm_lower_alarm_tempWstatus(lower_alarm_tempWstatus: Isolette_Data_Model::TempWstatus_i): Base_Types::Boolean :=
                GUMBO_Library::GUMBO__Library::Allowed_LowerAlarmTempWStatus(lower_alarm_tempWstatus)

            //derived from Allowed_UpperAlarmTemp from MMI
            def sysProp_I_Assm_upper_alarm_tempWstatus(upper_alarm_tempWStatus: Isolette_Data_Model::TempWstatus_i): Base_Types::Boolean :=
                GUMBO_Library::GUMBO__Library::Allowed_UpperAlarmTempWStatus(upper_alarm_tempWstatus)

        //MMI Compute Assumptions
            
            //----------------------------------------------
            // An integration constraint can only refer to a single port, so need a general assume clause
            // in order to relate the lower and uper temps
            //
            // derived from Allowed_AlarmTempWstatus_Ranges 
            //----------------------------------------------

            def sysProp_Allowed_AlarmTempWstatus_Ranges(lower_alarm_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                                        upper_alarm_tempWStatus: Isolette_Data_Model::TempWstatus_i): Base_Types::Boolean :=
                        GUMBO_Library::GUMBO__Library::Allowed_AlarmTempWStatus_Ranges(lower_alarm_tempWstatus, upper_alarm_tempWstatus);

        //MA Compute assumptions

            //----------------------------------------------
            // Range [96..101]
            // https://www.faa.gov/sites/faa.gov/files/aircraft/air_cert/design_approvals/air_software/AR-08-32.pdf#page=112  
            //
            // derived from Table_A_12_LowerAlarmTemp 
            //----------------------------------------------

            def sysProp_Table_A_12_LowerAlarmTemp(lower_alarm_temp: Isolette_Data_Model::Temp_i): Base_Types::Boolean :=
                        GUMBO_Library::GUMBO__Library::Allowed_LowerAlarmTemp(lower_alarm_temp.degrees);

            //----------------------------------------------
            // Range [97..102]
            // https://www.faa.gov/sites/faa.gov/files/aircraft/air_cert/design_approvals/air_software/AR-08-32.pdf#page=112
            //
            // derived from Table_A_12_UpperAlarmTemp 
            //----------------------------------------------

            def sysProp_Table_A_12_UpperAlarmTemp(upper_alarm_temp: Isolette_Data_Model::Temp_i): Base_Types::Boolean :=
                        GUMBO_Library::GUMBO__Library::Allowed_UpperAlarmTemp(upper_alarm_temp.degrees);

            //----------------------------------------------
            // This is a weakened for of the Figure_A_7 assumption for the MA
            //
            // derived from Figure_A_7_Weakened 
            //----------------------------------------------

            def sysProp_Figure_A_7_Weakened(lower_alarm_temp: Isolette_Data_Model::Temp_i, 
                                            upper_alarm_temp: Isolette_Data_Model::Temp_i): Base_Types::Boolean :=
                        lower_alarm_temp.degrees <= upper_alarm_temp.degrees;

    //Operator Interface

        //OI IC Guarantees

            //----------------------------------------------
            // Table_A_12_LowerAlarmTemp: Range [96..101]:
            // https://www.faa.gov/sites/faa.gov/files/aircraft/air_cert/design_approvals/air_software/AR-08-32.pdf#page=112
            //
            // derived from Allowed_LowerAlarmTempWstatus	
            //----------------------------------------------

            def sysProp_Allowed_LowerAlarmTempWstatus(lower_alarm_tempWstatus: Isolette_Data_Model::TempWstatus_i): Base_Types::Boolean :=
                GUMBO_Library::GUMBO__Library::Allowed_LowerAlarmTempWStatus(lower_alarm_tempWstatus);

            //----------------------------------------------
            // Table_A_12_UpperAlarmTemp: Range [97..102]
            // https://www.faa.gov/sites/faa.gov/files/aircraft/air_cert/design_approvals/air_software/AR-08-32.pdf#page=112
            //
            // derived from Allowed_UpperAlarmTempWstatus	
            //----------------------------------------------

            def sysProp_Allowed_UpperAlarmTempWstatus(upper_alarm_tempWStatus: Isolette_Data_Model::TempWstatus_i): Base_Types::Boolean :=
                GUMBO_Library::GUMBO__Library::Allowed_UpperAlarmTempWStatus(upper_alarm_tempWStatus);

        //OI Compute Guarantees

            //----------------------------------------------
            // An integration constraint can only refer to a single port, so need a general requires
            // clause to relate the lower and upper temps
            //
            // derived from Allowed_AlarmTempWStatus_Ranges
            //----------------------------------------------

            def sysProp_Allowed_AlarmTempWStatus_Ranges(lower_alarm_tempWstatus: Isolette_Data_Model::TempWstatus_i,
                                                        upper_alarm_tempWStatus: Isolette_Data_Model::TempWstatus_i): Base_Types::Boolean :=
                GUMBO_Library::GUMBO__Library::Allowed_AlarmTempWStatus_Ranges(lower_alarm_tempWstatus, upper_alarm_tempWstatus);
        
Contract {
	SysAssert {
        Name: START_Assert,
        Assert: True
    },
    SplitJoin {
        Schema { // OI
            SysAssert {
                Name: START_OI_Assert,
                Assert: True
            },
            Component { //assumptions proven
                Name: oi
            },
            SysAssert {
                Name: END_OI_Assert,
                Assert:
                    //derived from component contract
                    sysProp_Allowed_LowerAlarmTempWstatus(lower_alarm_tempWstatus)
                    and sysProp_Allowed_UpperAlarmTempWstatus(upper_alarm_tempWStatus)
                    and sysProp_Allowed_AlarmTempWStatus_Ranges(lower_alarm_tempWstatus, upper_alarm_tempWStatus)
                    and sysProp_lower_is_not_higher_than_upper(lower_desired_tempWstatus, upper_desired_tempWstatus)
            }
        },
        Schema { // TS
            SysAssert {
                Name: START_TS_Assert,
                Assert: True
            },
            Component { //assumptions proven
                Name: ts
            },
            SysAssert {
                Name: END_TS_Assert,
                Assert: True
            }
        }
    },
    SysAssert {
        Name: Post_TS_OI_Assert,
        Assert:
            //derived from JOIN
            sysProp_Allowed_LowerAlarmTempWstatus(lower_alarm_tempWstatus)
            and sysProp_Allowed_UpperAlarmTempWstatus(upper_alarm_tempWStatus)
            and sysProp_Allowed_AlarmTempWStatus_Ranges(lower_alarm_tempWstatus, upper_alarm_tempWStatus)
            and sysProp_lower_is_not_higher_than_upper(lower_desired_tempWstatus, upper_desired_tempWstatus)
    },
    SplitJoin {
        Schema { // Regulator
            SysAssert {
                Name: START_Regulator_Assert,
                Assert: 
                    //derived from SPLIT
                    sysProp_lower_is_not_higher_than_upper(lower_desired_tempWstatus, upper_desired_tempWstatus)
            },
            Component { //assumptions proven
                Name: drf
            },
            SysAssert {
                Name: Post_DRF_Assert,
                Assert: 
                    //holds from frame condition
                    sysProp_lower_is_not_higher_than_upper(lower_desired_tempWstatus, upper_desired_tempWstatus)
            },
            Component { //assumptions proven
                Name: mri
            },
            SysAssert {
                Name: Post_MRI_Assert,
                Assert:
                    // derived from component contract
                        sysProp_REQ_MRI_7(lower_desired_tempWstatus,  upper_desired_tempWstatus, interface_failure)
                        and sysProp_REQ_MRI_8(lower_desired_tempWstatus, upper_desired_tempWstatus, lower_desired_temp, upper_desired_temp, interface_failure)
                        and sysProp_lower_is_lower_temp(lower_desired_temp, upper_desired_temp)
            },
            Component { //assumptions proven
                Name: mrm
            },
            SysAssert {
                Name: Post_MRM_Assert,
                Assert:
                    //holds from frame condition
                    sysProp_REQ_MRI_7(lower_desired_tempWstatus,  upper_desired_tempWstatus, interface_failure)
                    and sysProp_REQ_MRI_8(lower_desired_tempWstatus, upper_desired_tempWstatus, lower_desired_temp, upper_desired_temp, interface_failure)
                    and sysProp_lower_is_lower_temp(lower_desired_temp, upper_desired_temp)
            },
            Component { //assumptions proven
                Name: mhs
            },
            SysAssert { //look at REQ_MHS_2
                Name: END_Regulator_Assert,
                Assert: 
                    sysProp_NormalModeHeatOnn(regulator_mode, currentTempWStatus, lowerDesiredTempWStatus, upperDesiredTempWStatus, internalFailure, heat_control)                   
            }
        },
        Schema { //Monitor
            SysAssert {
                Name: START_Monitor_Assert,
                Assert: 
                    //derived from SPLIT
                    sysProp_Allowed_LowerAlarmTempWstatus(lower_alarm_tempWstatus)
                    and sysProp_Allowed_UpperAlarmTempWstatus(upper_alarm_tempWStatus)
                    and sysProp_Allowed_AlarmTempWStatus_Ranges(lower_alarm_tempWstatus, upper_alarm_tempWStatus)
            },
            Component { //assumptions proven
                Name: dmf
            },
            SysAssert {
                Name: Post_DMF_Assert,
                Assert: 
                    //hold from frame condition
                    sysProp_Allowed_LowerAlarmTempWstatus(lower_alarm_tempWstatus)
                    and sysProp_Allowed_UpperAlarmTempWstatus(upper_alarm_tempWStatus)
                    and sysProp_Allowed_AlarmTempWStatus_Ranges(lower_alarm_tempWstatus, upper_alarm_tempWStatus)
            },
            Component { //assumptions proven
                Name: mmi
            },
            SysAssert {
                Name: Post_MMI_Assert,
                Assert: 
                    // Dervied from component contract
                        sysProp_REQ_MMI_5(lower_alarm_tempWstatus, upper_alarm_tempWstatus, interface_failure)
                        and sysProp_REQ_MMI_6(lower_alarm_tempWstatus, upper_alarm_tempWstatus, lower_alarm_temp, upper_alarm_temp, interface_failure)
                        and sysProp_Figure_A_7_Weakened(lower_alarm_temp, upper_alarm_temp)
                    // Derived from component contract, write frame, and pre-assertions
                        and sysProp_Table_A_12_LowerAlarmTemp(lower_alarm_temp)
                        and sysProp_Table_A_12_UpperAlarmTemp(upper_alarm_temp)
            },
            Component { //assumptions proven
                Name: mmm
            },
            SysAssert { 
                Name: Post_MMM_Assert,
                Assert: 
                    //These are preserved due to frame conditions
                    sysProp_REQ_MMI_5(lower_alarm_tempWstatus, upper_alarm_tempWstatus, interface_failure)
                    and sysProp_REQ_MMI_6(lower_alarm_tempWstatus, upper_alarm_tempWstatus, lower_alarm_temp, upper_alarm_temp, interface_failure)
                    and sysProp_Figure_A_7_Weakened(lower_alarm_temp, upper_alarm_temp)
                    and sysProp_Table_A_12_LowerAlarmTemp(lower_alarm_temp)
                    and sysProp_Table_A_12_UpperAlarmTemp(upper_alarm_temp)   
            },
            Component { //assumptions proven
                Name: ma
            },
            SysAssert { //look at REQ_MA_2
                Name: END_Monitor_Assert,
                Assert:
                    sysProp_NormalModeAlarmOn(lower_alarm_tempWstatus, upper_alarm_tempWStatus, current_tempWstatus, monitor_mode, interface_failure, alarm_control)
            }
        }
    }
    SysAssert { // HS
        Name: START_HS_Assert,
        Assert: True
    },
    Component { //assumptions proven
        Name: hs
    },
    SysAssert {
        Name: END_Assert,
        Assert: True
    }
}
